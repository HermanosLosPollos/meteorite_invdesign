import pandas as pd
from pymatgen.core import Composition, Element
from mendeleev import element as elem_data
from matminer.featurizers.composition import ElementProperty

# --- 1. SETUP CACHE ---
element_cache = {}

# --- 2. CLEANUP ---
cols_to_drop = ['MagpieData mean Density', 'Mean_EN', 'Max_EN', 'Mean_Rad',
                'Mean_Mass', 'Mean_Group', 'Mean_MP', 'Mean_BP', 'Mean_IE', 'Mean_EA']
my_samples = my_samples.drop(columns=[c for c in cols_to_drop if c in my_samples.columns])

# --- 3. CODE 1: MATMINER (Density) ---
ep = ElementProperty(data_source="magpie", features=["Density"], stats=["mean"])
my_samples = ep.featurize_dataframe(my_samples, col_id="composition")

# --- 4. CODE 2: PYMATGEN (EN, Radius, Group, AND Atomic Mass) ---
def get_pymatgen_features(formula):
    comp = Composition(formula)
    el_amt = comp.get_el_amt_dict()
    total_atoms = comp.num_atoms
    elements = {Element(symbol): amt for symbol, amt in el_amt.items()}

    mean_en = sum((getattr(el, "X", 0) or 0) * amt for el, amt in elements.items()) / total_atoms
    max_en = max((getattr(el, "X", 0) or 0) for el in elements.keys())
    mean_rad = sum((getattr(el, "atomic_radius", 0) or 0) * amt for el, amt in elements.items()) / total_atoms

    # NEW FEATURE: Atomic Mass (Replacing Covalent Radius)
    mean_mass = sum((getattr(el, "atomic_mass", 0) or 0) * amt for el, amt in elements.items()) / total_atoms

    mean_group = sum((getattr(el, "group", 0) or 0) * amt for el, amt in elements.items()) / total_atoms

    return pd.Series([mean_en, max_en, mean_rad, mean_mass, mean_group])

my_samples[['Mean_EN', 'Max_EN', 'Mean_Rad', 'Mean_Mass', 'Mean_Group']] = my_samples['formula'].apply(get_pymatgen_features)

# --- 5. MENDELEEV (Thermal & Electron Properties) ---
def get_mendeleev_features_fast(formula):
    comp = Composition(formula)
    el_amt = comp.get_el_amt_dict()
    total_atoms = comp.num_atoms
    mp, bp, ie, ea = 0, 0, 0, 0
    for symbol, amt in el_amt.items():
        if symbol not in element_cache:
            element_cache[symbol] = elem_data(symbol)
        e = element_cache[symbol]
        mp += (getattr(e, "melting_point", 0) or 0) * amt
        bp += (getattr(e, "boiling_point", 0) or 0) * amt
        i_energies = getattr(e, "ionenergies", {})
        ie += (i_energies[1] if (i_energies and 1 in i_energies) else 0) * amt
        ea += (getattr(e, "electron_affinity", 0) or 0) * amt
    return pd.Series([mp/total_atoms, bp/total_atoms, ie/total_atoms, ea/total_atoms])

my_samples[['Mean_MP', 'Mean_BP', 'Mean_IE', 'Mean_EA']] = my_samples['formula'].apply(get_mendeleev_features_fast)

# --- 6. FINAL SELECTION ---
final_cols = ['name', 'formula', 'Mean_EN', 'Max_EN', 'Mean_Rad', 'Mean_Mass',
              'Mean_MP', 'Mean_BP', 'MagpieData mean Density',
              'Mean_IE', 'Mean_EA', 'Mean_Group']

my_final_10 = my_samples[final_cols]
print("SUCCESS: 10 valid features extracted.")
print(my_final_10.head())
